Rowset

The rowset is an object which holds a collection of row objects.
You can add extra functions to this object to give extra functionality.

Below I will show a function which will add a category to all the columns when a category_id is found.
However this can better be set in a table behavior.

{% highlight javascript %} var Rowset = require('raddish').Rowset; var util = require('util'); function DemoRowset(config) { Rowset.call(this, config); } util.inherits(DemoRowset, Rowset); DemoRowset.prototype.setCategory = function() { var self = this; for(var index in this.rows) { var row = this.row[index]; if(row.getData().category_id) { this.getObject('com://demo/foo.model.category') .then(function(model) { return model.set('id', row.getData().category_id).getItem(); }) .then(function(category) { row.category = category; });; } } return this.rows; }; module.exports = DemoRowset; {% endhighlight %}
